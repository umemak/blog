RUN cp -r ./{lib,share,include,bin} /usr

でファイル未存在のエラーになってしまった。

RUN cp -r ./lib /usr
RUN cp -r ./share /usr
RUN cp -r ./include /usr
RUN cp -r ./bin /usr
